            \hline
            Ley de Ampère-Maxwell (int.) & $\oint \vec{H} \cdot d\vec{l}' = I + \dfrac{d}{dt} \int \vec{D} \cdot d\vec{S}'$ \\
            \hline
            Ley de Ampère (conductores)  & $\nabla \times \vec{H} = \vec{J} + \sigma \vec{E}$                               \\
            \hline
        \end{tabular}
    }
    \caption{Fórmulas de las leyes de Maxwell}
\end{table}
